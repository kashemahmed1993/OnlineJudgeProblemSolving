#include <iostream>
#include <cstdio>
#include <cstring>

#define N 50

using namespace std;

int n,t,arr[N],tmp,flag,len,noError;

void resetArray(){
    for(int i=0;i<N;i++){
        arr[i]=0;
    }
}

int main() {
    char ch[2000];
    while(scanf("%d",&t)==1){
        for(int i=0;i<t;i++)
        {
            resetArray();
            scanf("%d",&n);
            scanf("%s",ch);
            flag=0;
            noError=1;
            len = strlen(ch);
            for(int i=0;i<len;i++){
                if(ch[i]!='_'){
                    tmp=ch[i]-'A';
                    arr[tmp]++;
                    if(i-1>=0&&i+1<len){
                        if(ch[i-1]!=ch[i]&&ch[i]!=ch[i+1]){
                            noError=0;
                        }
                    } else if(i+1<len){
                        if(ch[i]!=ch[i+1]){
                            noError=0;
                        }
                    } else{
                        if(ch[i]!=ch[i-1]){
                            noError=0;
                        }
                    }
                    
                }else{
                    flag=1;
                }
            }
            if(!noError)
            for(int i=0;i<N;i++){
                if(!arr[i])
                    continue;
                if(arr[i]==1){
                    flag=0;
                    break;
                }
                else {
                    if(!flag){
                        break;
                    }
                }
            }
            if (noError){
                flag =1;
            }
            if(flag){
                printf("YES\n");
            }
            else {
                printf("NO\n");
            }
        }
    }
    return 0;
}
