#include <iostream>
#include <cstdio>
#include <cstring>

#define N 50

using namespace std;


int main(){
    int t,n,a,b,sum,mx,mn;
    while(scanf("%d",&t)==1){
        for(int j=0;j<t;j++){
            scanf("%d %d %d",&n,&a,&b);
            sum=0;
            if (a!=b){
                mx=a>b?a:b;
                mn=a<b?a:b;
                for(int i=1;i<n;i++)
                {
                    sum+=mn;
                }
                printf("%d ",sum);
                for(int i=n-1;i>0;i--){
                    sum=sum-mn+mx;
                    printf("%d",sum);
                    if(i!=1){
                        printf(" ");
                    }
                }
                printf("\n");
                
            }
            else {
                for(int i=1;i<n;i++)
                {
                    sum=sum+a;
                }
                printf("%d\n",sum);
            }
            
        }
    }
}
