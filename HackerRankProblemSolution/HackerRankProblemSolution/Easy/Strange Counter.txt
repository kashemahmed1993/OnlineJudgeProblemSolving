#include <iostream>
#include <cstdio>
#define N 50
#define ll long long
using namespace std;

ll arr[N],n;

void generateArray(){
    arr[0]=3;
    for(int i=1;i<N;i++)
        arr[i]=arr[i-1]<<1;
}
   

int main() {
    generateArray();
    while(scanf("%lld",&n)==1){
        int i=0;
        while(n>arr[i]){
            n=n-arr[i++];
        }
        n=arr[i]-n+1;
        printf("%lld\n",n);
    }
    return 0;
}
